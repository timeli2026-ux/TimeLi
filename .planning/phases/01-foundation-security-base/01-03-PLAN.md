<task type="auto">
  <name>Task 1: Create environment variable validation</name>
  <files>src/lib/env.ts</files>
  <action>
    Create src/lib/env.ts that validates all required environment variables at import time:

    ```typescript
    import { z } from 'zod'

    const envSchema = z.object({
      // Supabase (public - safe to expose)
      NEXT_PUBLIC_SUPABASE_URL: z.string().url(),
      NEXT_PUBLIC_SUPABASE_ANON_KEY: z.string().min(1),

      // Supabase (private - server only)
      SUPABASE_SERVICE_ROLE_KEY: z.string().min(1).optional(),

      // Node environment
      NODE_ENV: z.enum(['development', 'production', 'test']).default('development'),
    })

    // Validate on import - fail fast if missing
    function validateEnv() {
      const parsed = envSchema.safeParse(process.env)

      if (!parsed.success) {
        console.error('❌ Invalid environment variables:')
        console.error(parsed.error.flatten().fieldErrors)
        throw new Error('Invalid environment variables')
      }

      return parsed.data
    }

    export const env = validateEnv()

    // Type-safe env access
    export type Env = z.infer<typeof envSchema>
    ```

    Install zod if not present:
    ```bash
    npm install zod
    ```

    This ensures the app fails fast on startup if required env vars are missing, rather than failing at runtime when the var is accessed.
  </action>
  <verify>npm run build succeeds (will fail if env vars missing - expected in CI until configured)</verify>
  <done>Environment validation module created with Zod schema</done>
</task>

<task type="auto">
  <name>Task 2: Set up pre-commit hooks for secrets scanning</name>
  <files>.husky/pre-commit, .gitleaks.toml, package.json</files>
  <action>
    Install husky and gitleaks for pre-commit secret scanning:
    ```bash
    npm install -D husky
    npx husky init
    ```

    Create .gitleaks.toml configuration:
    ```toml
    title = "TimeLi Gitleaks Config"

    [allowlist]
    description = "Allowlist for non-sensitive patterns"
    paths = [
      '''\.env\.example''',
      '''\.env\.local\.example''',
    ]

    [[rules]]
    id = "supabase-anon-key"
    description = "Supabase Anon Key"
    regex = '''eyJ[A-Za-z0-9-_]+\.eyJ[A-Za-z0-9-_]+'''
    tags = ["supabase", "api-key"]

    [[rules]]
    id = "generic-api-key"
    description = "Generic API Key"
    regex = '''(?i)(api[_-]?key|apikey|secret[_-]?key)\s*[:=]\s*['"][A-Za-z0-9-_]{20,}['"]'''
    tags = ["api-key"]
    ```

    Update .husky/pre-commit to run gitleaks:
    ```bash
    #!/usr/bin/env sh
    . "$(dirname -- "$0")/_/husky.sh"

    # Run gitleaks if available
    if command -v gitleaks &> /dev/null; then
      gitleaks protect --staged --verbose
    else
      echo "⚠️  gitleaks not installed - skipping secrets scan"
      echo "   Install with: brew install gitleaks"
    fi

    # Run TypeScript check
    npm run build --if-present
    ```

    Add npm script for manual scanning in package.json:
    ```json
    "scripts": {
      "secrets:scan": "gitleaks detect --verbose"
    }
    ```

    Note: gitleaks binary must be installed separately (brew install gitleaks or from releases).
    The pre-commit hook gracefully skips if gitleaks is not installed.
  </action>
  <verify>.husky/pre-commit exists and is executable, .gitleaks.toml exists</verify>
  <done>Pre-commit hook configured for secrets scanning</done>
</task>
