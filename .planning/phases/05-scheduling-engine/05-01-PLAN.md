Algorithm:
```typescript
function generateSchedule(input: SchedulerInput): SchedulerResult {
  // 1. Build blocked slots from preferences (with weekend sleep support)
  const blockedSlots = buildBlockedSlots(input)

  // 2. Initialize schedule with fixed events (meals, sleep, commitments, commute)
  const fixedEvents = createFixedEvents(input, blockedSlots)

  // 3. Separate anchored goals from regular goals
  const { anchoredGoals, regularGoals } = partitionGoals(input.goals)

  // 4. Schedule anchored goals FIRST (habit stacking)
  // They MUST be adjacent to their anchor events
  const anchoredEvents = scheduleAnchoredGoals(anchoredGoals, input.commitments, fixedEvents)

  // 5. Sort remaining goals by MRV heuristic (most constrained first)
  // NOT by hours_per_week - MRV finds failures faster
  const sortedGoals = sortByConstraintTightness(regularGoals, availableSlots)

  // 6. Add deadline-based priority boost
  const prioritizedGoals = applyDeadlinePriority(sortedGoals, input.weekStart)

  // 7. Use backtracking CSP solver for regular goals
  const solution = backtrackSolve(prioritizedGoals, availableSlots, [...fixedEvents, ...anchoredEvents])

  // 8. Calculate stats and return
  return buildResult(solution, input)
}
```

**Backtracking Solver:**
```typescript
const MAX_BACKTRACKS = 1000 // prevent infinite search

function backtrackSolve(
  remainingGoals: GoalWithMetadata[],
  availableSlots: Map<number, TimeSlot[]>,
  assignments: ScheduleEvent[],
  backtrackCount: { count: number }
): ScheduleEvent[] | null {

  if (remainingGoals.length === 0) {
    return assignments // All goals scheduled successfully
  }

  if (backtrackCount.count++ > MAX_BACKTRACKS) {
    return null // Exceeded search budget, will report partial solution
  }

  const goal = remainingGoals[0]
  const sessionsNeeded = calculateSessionsNeeded(goal)

  // Get all valid slot combinations for this goal's sessions
  // Distribute across different days when possible (spaced practice)
  const validPlacements = findAllValidPlacements(goal, sessionsNeeded, availableSlots, assignments)

  // Try each placement (will be ranked by scoring in Plan 02)
  for (const placement of validPlacements) {
    const newAvailable = subtractSlots(availableSlots, placement)
    const newAssignments = [...assignments, ...placementToEvents(goal, placement)]

    const result = backtrackSolve(
      remainingGoals.slice(1),
      newAvailable,
      newAssignments,
      backtrackCount
    )

    if (result !== null) return result
  }

  return null // Backtrack - no valid placement for this goal
}
```

**Habit Stacking (Implementation Intentions):**
```typescript
function scheduleAnchoredGoals(
  anchoredGoals: GoalWithMetadata[],
  commitments: FixedCommitment[],
  existingEvents: ScheduleEvent[]
): ScheduleEvent[] {
  const events: ScheduleEvent[] = []

  for (const goal of anchoredGoals) {
    if (!goal.anchor) continue

    // Find all instances of anchor event in the week
    const anchorInstances = commitments.filter(c => c.id === goal.anchor.anchorId)

    for (const anchor of anchorInstances) {
      // Calculate slot immediately after/before anchor
      const slot = goal.anchor.type === 'after_event'
        ? createSlotAfter(anchor, goal.anchor.bufferMinutes, goal.sessionStrategy.preferredDuration)
        : createSlotBefore(anchor, goal.anchor.bufferMinutes, goal.sessionStrategy.preferredDuration)

      if (canPlaceEvent(slot, [...existingEvents, ...events], 0)) {
        events.push({
          id: generateId(),
          type: 'goal',
          title: goal.title,
          slot,
          goalId: goal.id,
          realmId: goal.realmId,
          isLocked: false,
          cognitiveLoad: goal.cognitiveLoad,
          isAnchoredSession: true
        })
      }
    }
  }

  return events
}
```

**MRV Heuristic (Most Restricted Variable):**
```typescript
function sortByConstraintTightness(goals: GoalWithMetadata[], slots: Map<number, TimeSlot[]>): GoalWithMetadata[] {
  return goals.sort((a, b) => {
    const aOptions = countValidPlacements(a, slots)
    const bOptions = countValidPlacements(b, slots)
    return aOptions - bOptions // Fewer options = more constrained = schedule first
  })
}
```
