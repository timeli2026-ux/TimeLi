<architecture_decision>
## Self-Hosted LLM Strategy

**Cost model:** GPU compute billed per hour while VM is on.
**Approach:** Self-host vLLM on RunPod/Lambda Labs, run during fixed daily windows.
**Fallback:** Graceful degradation when LLM is offline (disabled chat, no API fallback).

**Provider routing (server-side, transparent to user):**
1. Health check self-hosted endpoint with 2s timeout
2. If available → route to vLLM
3. If unavailable → return "offline" status, UI shows disabled state

**Why not API fallback:** Early-stage cost optimization. User prefers predictable GPU costs over per-request API billing.
</architecture_decision>

```typescript
import type { LLMProvider, ChatMessage, ChatOptions, ChatResponse, LLMStatus } from '../types'

export class SelfHostedProvider implements LLMProvider {
  private baseUrl: string
  private model: string
  private availabilityCache: { available: boolean; checkedAt: number } | null = null
  private readonly CACHE_TTL_MS = 30000  // Cache health status for 30s

  constructor(baseUrl: string, model: string) {
    this.baseUrl = baseUrl.replace(/\/$/, '')  // Remove trailing slash
    this.model = model
  }

  getName(): string {
    return 'self-hosted-vllm'
  }

  async isAvailable(): Promise<boolean> {
    // Return cached result if fresh
    if (this.availabilityCache && Date.now() - this.availabilityCache.checkedAt < this.CACHE_TTL_MS) {
      return this.availabilityCache.available
    }

    try {
      const controller = new AbortController()
      const timeoutId = setTimeout(() => controller.abort(), 2000)  // 2s timeout

      const res = await fetch(`${this.baseUrl}/health`, {
        signal: controller.signal,
      })
      clearTimeout(timeoutId)

      const available = res.ok
      this.availabilityCache = { available, checkedAt: Date.now() }
      return available
    } catch {
      this.availabilityCache = { available: false, checkedAt: Date.now() }
      return false
    }
  }

  async chat(messages: ChatMessage[], options?: ChatOptions): Promise<ChatResponse> {
    // OpenAI-compatible API (vLLM exposes this format)
    const response = await fetch(`${this.baseUrl}/chat/completions`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: this.model,
        messages: messages.map(m => ({ role: m.role, content: m.content })),
        max_tokens: options?.maxTokens ?? 500,
        temperature: options?.temperature ?? 0.7,
      }),
    })

    if (!response.ok) {
      const error = await response.text()
      throw new Error(`LLM request failed: ${response.status} ${error}`)
    }

    const data = await response.json()
    return {
      content: data.choices[0].message.content,
      provider: this.getName(),
      tokensUsed: data.usage?.total_tokens,
    }
  }

  async getStatus(): Promise<LLMStatus> {
    const available = await this.isAvailable()
    return {
      available,
      provider: this.getName(),
      message: available ? undefined : 'Chat is currently offline',
    }
  }
}
```

Key features:
- Health check with 2s timeout (doesn't block requests)
- 30s availability cache (reduces health check calls)
- OpenAI-compatible API format (works with vLLM, text-generation-inference, ollama)

export class OfflineProvider implements LLMProvider {
  private offlineMessage: string

  constructor(offlineMessage?: string) {
    this.offlineMessage = offlineMessage ?? 'Chat is currently unavailable. Please try again later.'
  }

  getName(): string {
    return 'offline'
  }

  async isAvailable(): Promise<boolean> {
    return true  // Always "available" as a fallback
  }

  async chat(_messages: ChatMessage[], _options?: ChatOptions): Promise<ChatResponse> {
    // Don't actually process - return offline message
    return {
      content: this.offlineMessage,
      provider: this.getName(),
    }
  }

  async getStatus(): Promise<LLMStatus> {
    return {
      available: false,  // Report as unavailable so UI can show disabled state
      provider: this.getName(),
      message: this.offlineMessage,
    }
  }
}
```

This provider is used when self-hosted LLM is down. It doesn't call any API - just returns the offline message.

import { SelfHostedProvider } from './providers/self-hosted'
import { OfflineProvider } from './providers/offline'
import type { LLMProvider, LLMStatus } from './types'

let cachedProvider: LLMProvider | null = null
let providerResolvedAt = 0
const PROVIDER_CACHE_TTL_MS = 30000  // Re-check every 30s

export async function getLLMProvider(): Promise<LLMProvider> {
  // Return cached if fresh
  if (cachedProvider && Date.now() - providerResolvedAt < PROVIDER_CACHE_TTL_MS) {
    return cachedProvider
  }

  const baseUrl = process.env.LLM_BASE_URL
  const model = process.env.LLM_MODEL

  // If not configured, return offline provider
  if (!baseUrl || !model) {
    cachedProvider = new OfflineProvider('Chat is not configured. LLM_BASE_URL and LLM_MODEL required.')
    providerResolvedAt = Date.now()
    return cachedProvider
  }

  const selfHosted = new SelfHostedProvider(baseUrl, model)

  if (await selfHosted.isAvailable()) {
    cachedProvider = selfHosted
    providerResolvedAt = Date.now()
    return cachedProvider
  }

  // Self-hosted is down, use offline provider
  cachedProvider = new OfflineProvider()
  providerResolvedAt = Date.now()
  return cachedProvider
}

export async function getLLMStatus(): Promise<LLMStatus> {
  const provider = await getLLMProvider()
  return provider.getStatus()
}
```

Create POST /api/chat endpoint:

1. Authenticate user (same pattern as schedule endpoints)
2. Parse body: { message: string, weekStart: string }
3. Check LLM availability first (fast fail if offline)
4. Load conversation history using conversation service
5. Build system prompt for schedule assistant
6. Call LLM via router
7. Save assistant response to conversation
8. Return response to client

```typescript
import { createServerClient } from '@/lib/supabase/server'
import { getLLMProvider, getLLMStatus } from '@/lib/llm'
import { getOrCreateConversation, addMessage, toPromptMessages } from '@/lib/services/conversation'
import { rateLimit } from '@/lib/rate-limit'
import { NextResponse } from 'next/server'

export async function POST(request: Request) {
  // Rate limit
  const rateLimitResult = await rateLimit(request, { limit: 20, window: 60 })
  if (rateLimitResult) return rateLimitResult

  // Auth
  const supabase = await createServerClient()
  const { data: { user }, error: authError } = await supabase.auth.getUser()
  if (authError || !user) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  // Parse body
  const body = await request.json()
  const { message, weekStart } = body
  if (!message || !weekStart) {
    return NextResponse.json({ error: 'message and weekStart required' }, { status: 400 })
  }

  // Check LLM status first (fast fail)
  const status = await getLLMStatus()
  if (!status.available) {
    return NextResponse.json({
      error: 'Chat unavailable',
      offline: true,
      message: status.message,
    }, { status: 503 })
  }

  // Get provider and conversation
  const provider = await getLLMProvider()
  const conversation = await getOrCreateConversation(supabase, user.id, weekStart)

  // Add user message
  const userMessage = { role: 'user' as const, content: message, timestamp: new Date().toISOString() }
  await addMessage(supabase, conversation.id, userMessage)

  // Build prompt with conversation history
  const systemPrompt = `You are a helpful schedule assistant for TimeLi.
Help the user understand and modify their weekly schedule.
Keep responses concise (this is chat, not essay format).
If the user asks to modify their schedule, acknowledge and confirm the change.`

  const messages = toPromptMessages([...conversation.messages, userMessage], systemPrompt)

  // Call LLM
  const response = await provider.chat(messages, { maxTokens: 500 })

  // Save assistant response
  const assistantMessage = { role: 'assistant' as const, content: response.content, timestamp: new Date().toISOString() }
  await addMessage(supabase, conversation.id, assistantMessage)

  return NextResponse.json({
    message: response.content,
    provider: response.provider,
  })
}
```

Handle errors gracefully - return user-friendly messages, never expose internal errors.
