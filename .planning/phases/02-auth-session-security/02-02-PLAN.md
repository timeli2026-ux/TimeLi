```typescript
// src/lib/supabase/oauth.ts
import { createClient } from '@/lib/supabase/client'

export async function signInWithGoogle() {
  const supabase = createClient()

  const { error } = await supabase.auth.signInWithOAuth({
    provider: 'google',
    options: {
      redirectTo: `${window.location.origin}/auth/callback`,
      queryParams: {
        access_type: 'offline',
        prompt: 'consent',
      },
    },
  })

  if (error) {
    throw error
  }
}
```

The helper:
- Uses browser Supabase client
- Configures redirectTo to use the existing auth callback route
- Requests offline access for refresh tokens
- Prompts for consent to ensure we get refresh token
